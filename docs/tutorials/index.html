Static Website Server Beginner

Learn how to serve a static website with Rusty Beam.

Configuration
<!DOCTYPE html>
<html>
<body>
    <table itemscope itemtype="http://rustybeam.net/ServerConfig">
        <tr><td>Port</td><td itemprop="port">8080</td></tr>
        <tr><td>Root</td><td itemprop="root">./website</td></tr>
    </table>
    
    <ul>
        <li itemscope itemtype="http://rustybeam.net/HostConfig">
            <span itemprop="host">*</span>
            <ul>
                <!-- Compression for better performance -->
                <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
                    <span itemprop="library">file://./plugins/lib/compression.so</span>
                </li>
                <!-- File serving -->
                <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
                    <span itemprop="library">file://./plugins/lib/file-handler.so</span>
                </li>
            </ul>
        </li>
    </ul>
</body>
</html>
Usage
# Create website directory
mkdir -p website
echo "<h1>Welcome!</h1>" > website/index.html

# Start server
rusty-beam static-config.html

# Access your site
curl http://localhost:8080/

Dynamic Content Management Beginner

Use CSS selectors to dynamically update HTML content.

Initial HTML Page
<!DOCTYPE html>
<html>
<head><title>Dynamic Page</title></head>
<body>
    <header>
        <h1 id="title">Original Title</h1>
        <nav id="menu">
            <ul>
                <li><a href="/">Home</a></li>
            </ul>
        </nav>
    </header>
    <main id="content">
        <p>Original content</p>
    </main>
</body>
</html>
Update Operations
# Update the title
curl -X PUT -H "Range: selector=#title" \
  -d '<h1 id="title">New Dynamic Title</h1>' \
  http://localhost:3000/page.html

# Add menu items
curl -X POST -H "Range: selector=#menu ul" \
  -d '<li><a href="/about">About</a></li>' \
  http://localhost:3000/page.html

# Replace main content
curl -X PUT -H "Range: selector=#content" \
  -d '<main id="content"><h2>Updated!</h2><p>New content here</p></main>' \
  http://localhost:3000/page.html

RESTful API with CORS Intermediate

Build a RESTful API with proper CORS configuration for frontend apps.

API Configuration
<li itemscope itemtype="http://rustybeam.net/HostConfig">
    <span itemprop="host">api.example.com</span>
    <span itemprop="root">./api-data</span>
    
    <ul>
        <!-- CORS configuration -->
        <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
            <span itemprop="library">file://./plugins/lib/cors.so</span>
            <meta itemprop="allowed-origins" content="https://app.example.com">
            <meta itemprop="allowed-methods" content="GET,POST,PUT,DELETE,OPTIONS">
            <meta itemprop="allowed-headers" content="Content-Type,Authorization">
            <meta itemprop="expose-headers" content="X-Total-Count">
            <meta itemprop="credentials" content="true">
            <meta itemprop="max-age" content="3600">
        </li>
        
        <!-- Rate limiting -->
        <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
            <span itemprop="library">file://./plugins/lib/rate-limit.so</span>
            <meta itemprop="requests-per-minute" content="60">
        </li>
        
        <!-- Selector handler for JSON manipulation -->
        <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
            <span itemprop="library">file://./plugins/lib/selector-handler.so</span>
        </li>
        
        <!-- File handler -->
        <li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
            <span itemprop="library">file://./plugins/lib/file-handler.so</span>
        </li>
    </ul>
</li>
Frontend JavaScript Client
class RustyBeamAPI {
    constructor(baseURL) {
        this.baseURL = baseURL;
    }
    
    async getUsers() {
        const response = await fetch(`${this.baseURL}/users.json`);
        return response.json();
    }
    
    async updateUser(userId, data) {
        const response = await fetch(
            `${this.baseURL}/users.json`,
            {
                method: 'PUT',
                headers: {
                    'Content-Type': 'application/json',
                    'Range': `selector=#user-${userId}`
                },
                body: JSON.stringify(data),
                credentials: 'include'
            }
        );
        return response.text();
    }
    
    async createUser(data) {
        const response = await fetch(
            `${this.baseURL}/users.json`,
            {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Range': 'selector=#users-list'
                },
                body: JSON.stringify(data),
                credentials: 'include'
            }
        );
        return response.text();
    }
}

// Usage
const api = new RustyBeamAPI('https://api.example.com');
const users = await api.getUsers();

Real-time Applications

Live Dashboard with WebSockets Advanced

Create a real-time dashboard using the WebSocket plugin.

Server Configuration
<li itemprop="plugin" itemscope itemtype="http://rustybeam.net/Plugin">
    <span itemprop="library">file://./plugins/lib/websocket.so</span>
    <meta itemprop="path" content="/ws">
    <meta itemprop="broadcast-changes" content="true">
</li>
Dashboard HTML
<!DOCTYPE html>
<html>
<head>
    <title>Live Dashboard</title>
    <style>
        .metric { 
            display: inline-block; 
            padding: 20px; 
            margin: 10px;
            background: #f0f0f0; 
            border-radius: 5px; 
        }
        .metric-value { 
            font-size: 2em; 
            font-weight: bold; 
        }
    </style>
</head>
<body>
    <h1>Real-Time Metrics</h1>
    
    <div id="metrics">
        <div class="metric">
            <div>CPU Usage</div>
            <div class="metric-value" id="cpu">0%</div>
        </div>
        <div class="metric">
            <div>Memory</div>
            <div class="metric-value" id="memory">0%</div>
        </div>
        <div class="metric">
            <div>Requests/sec</div>
            <div class="metric-value" id="requests">0</div>
        </div>
    </div>
    
    <div id="log">
        <h2>Event Log</h2>
        <ul id="events"></ul>
    </div>
    
    <script>
    // WebSocket connection
    const ws = new WebSocket('ws://localhost:3000/ws');
    
    ws.onopen = () => {
        console.log('Connected to dashboard');
        addEvent('Connected to server');
    };
    
    ws.onmessage = (event) => {
        const data = JSON.parse(event.data);
        
        if (data.type === 'file_changed' && data.path === '/metrics.json') {
            updateMetrics();
        }
    };
    
    ws.onerror = (error) => {
        console.error('WebSocket error:', error);
        addEvent('Connection error');
    };
    
    async function updateMetrics() {
        try {
            const response = await fetch('/metrics.json');
            const metrics = await response.json();
            
            document.getElementById('cpu').textContent = metrics.cpu + '%';
            document.getElementById('memory').textContent = metrics.memory + '%';
            document.getElementById('requests').textContent = metrics.requests;
            
            addEvent('Metrics updated');
        } catch (error) {
            console.error('Failed to fetch metrics:', error);
        }
    }
    
    function addEvent(message) {
        const events = document.getElementById('events');
        const li = document.createElement('li');
        li.textContent = `${new Date().toLocaleTimeString()}: ${message}`;
        events.insertBefore(li, events.firstChild);
        
        // Keep only last 10 events
        while (events.children.length > 10) {
            events.removeChild(events.lastChild);
        }
    }
    
    // Initial load
    updateMetrics();
    
    // Update metrics periodically as fallback
    setInterval(updateMetrics, 5000);
    </script>
</body>
</html>
Metrics Update Script
#!/bin/bash
# update-metrics.sh - Run this to update dashboard

while true; do
    CPU=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)
    MEM=$(free | grep Mem | awk '{print int($3/$2 * 100)}')
    REQUESTS=$(shuf -i 50-200 -n 1)  # Simulated
    
    cat > metrics.json << EOF
{
    "cpu": $CPU,
    "memory": $MEM,
    "requests": $REQUESTS,
    "timestamp": "$(date -Iseconds)"
}
EOF
    
    # Upload to server (triggers WebSocket broadcast)
    curl -X PUT -H "Content-Type: application/json" \
        --data-binary @metrics.json \
        http://localhost:3000/metrics.json
    
    sleep 2
done

Add New Post Script
#!/bin/bash
# add-post.sh - Add a new blog post

TITLE="$1"
CONTENT="$2"
DATE=$(date +"%Y-%m-%d")
ID=$(date +%s)

POST_HTML="<article id=\"post-$ID\">
    <h2>$TITLE</h2>
    <div class=\"meta\">Posted on $DATE</div>
    <div class=\"content\">
        $CONTENT
    </div>
</article>"

# Add post to blog (prepend to show newest first)
curl -X POST -H "Range: selector=#posts" \
    -H "Content-Type: text/html" \
    -d "$POST_HTML" \
    http://localhost:3000/blog.html

echo "Post added with ID: post-$ID"
Update Post Script
#!/bin/bash
# update-post.sh - Update an existing post

POST_ID="$1"
NEW_CONTENT="$2"

curl -X PUT -H "Range: selector=#$POST_ID .content" \
    -H "Content-Type: text/html" \
    -d "<div class=\"content\">$NEW_CONTENT</div>" \
    http://localhost:3000/blog.html
Delete Post Script
#!/bin/bash
# delete-post.sh - Delete a post

POST_ID="$1"

curl -X DELETE -H "Range: selector=#$POST_ID" \
    http://localhost:3000/blog.html

A/B Testing Framework Advanced

Implement A/B testing using selectors and cookies.

A/B Test Page
<!DOCTYPE html>
<html>
<body>
    <div id="hero">
        <h1 class="variant-a">Welcome to Our Service!</h1>
        <h1 class="variant-b" style="display:none">Start Your Journey Today!</h1>
        
        <button class="variant-a cta">Get Started</button>
        <button class="variant-b cta" style="display:none">Try It Free</button>
    </div>
    
    <script>
    // Simple A/B test implementation
    function getVariant() {
        const variant = localStorage.getItem('ab-variant');
        if (!variant) {
            const newVariant = Math.random() < 0.5 ? 'a' : 'b';
            localStorage.setItem('ab-variant', newVariant);
            return newVariant;
        }
        return variant;
    }
    
    function showVariant(variant) {
        document.querySelectorAll('.variant-a').forEach(el => {
            el.style.display = variant === 'a' ? '' : 'none';
        });
        document.querySelectorAll('.variant-b').forEach(el => {
            el.style.display = variant === 'b' ? '' : 'none';
        });
    }
    
    // Track clicks
    document.querySelectorAll('.cta').forEach(button => {
        button.addEventListener('click', () => {
            const variant = getVariant();
            // Send tracking data
            fetch('/track', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    event: 'cta_click',
                    variant: variant,
                    timestamp: new Date().toISOString()
                })
            });
        });
    });
    
    // Apply variant
    showVariant(getVariant());
    </script>
</body>
</html>

Best Practices

Development Tips

  • Start Simple: Begin with basic file serving before adding complexity
  • Test Incrementally: Test each plugin as you add it
  • Use Verbose Mode: Run with -v flag during development
  • Monitor Performance: Use the access-log plugin to track requests
  • Security First: Always use authentication for sensitive operations
